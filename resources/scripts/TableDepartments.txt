CREATE TABLE Departments (
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY, -- Tự động sinh khóa chính
    Name VARCHAR2(255) NOT NULL, -- Tên phòng ban
    Priority NUMBER(5) NOT NULL, -- Độ ưu tiên
    CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- Thời gian tạo
    UpdatedAt TIMESTAMP NULL, -- Thời gian cập nhật (nullable)
    PRIMARY KEY (Id) -- Khóa chính
);
